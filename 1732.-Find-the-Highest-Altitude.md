### 1732. Find the Highest Altitude

![image](https://user-images.githubusercontent.com/91203793/230568954-d3db5cc9-691a-4ae0-8d37-1261cfba6ad3.png)


### Solution

class Solution {
    public int largestAltitude(int[] gain) {
        int sum = 0;
        int greatest = 0;
        for(int i=0;i<gain.length;i++)
        {
            sum += gain[i];
            if(greatest < sum)
            {
                greatest = sum;
            }
        }
        return greatest;
    }
}
